        function extractVideoId(url) {
            const patterns = [
                /(?:youtube\.com\/watch\?v=)([^&\?\/]+)/,
                /(?:youtu\.be\/)([^&\?\/]+)/,
                /(?:youtube\.com\/embed\/)([^&\?\/]+)/,
                /(?:youtube\.com\/v\/)([^&\?\/]+)/,
                /^([a-zA-Z0-9_-]{11})$/
            ];

            for (const pattern of patterns) {
                const match = url.match(pattern);
                if (match && match[1]) return match[1];
            }
            return null;
        }

        function getThumbnails() {
            const urlInput = document.getElementById('youtubeUrl');
            const url = urlInput.value.trim();
            const errorMessage = document.getElementById('errorMessage');
            const thumbnailsSection = document.getElementById('thumbnailsSection');
            const thumbnailGrid = document.getElementById('thumbnailGrid');

            errorMessage.classList.remove('active');
            thumbnailsSection.classList.remove('active');

            if (!url) {
                showError('Please enter a YouTube URL');
                return;
            }

            const videoId = extractVideoId(url);

            if (!videoId) {
                showError('Invalid YouTube URL. Please check the URL and try again.');
                return;
            }

            const thumbnails = [
                {
                    quality: 'Maximum Resolution',
                    badge: 'HD',
                    resolution: '1280 x 720',
                    url: `https://i.ytimg.com/vi/${videoId}/maxresdefault.jpg`,
                    fallback: `https://img.youtube.com/vi/${videoId}/hqdefault.jpg`,
                    filename: `${videoId}_maxres.jpg`
                },
                {
                    quality: 'High Quality',
                    badge: 'HD',
                    resolution: '480 x 360',
                    url: `https://i.ytimg.com/vi/${videoId}/hqdefault.jpg`,
                    fallback: null,
                    filename: `${videoId}_hq.jpg`
                },
                {
                    quality: 'Medium Quality',
                    badge: 'SD',
                    resolution: '320 x 180',
                    url: `https://i.ytimg.com/vi/${videoId}/mqdefault.jpg`,
                    fallback: null,
                    filename: `${videoId}_mq.jpg`
                },
                {
                    quality: 'Standard Definition',
                    badge: 'SD',
                    resolution: '120 x 90',
                    url: `https://i.ytimg.com/vi/${videoId}/default.jpg`,
                    fallback: null,
                    filename: `${videoId}_sd.jpg`
                }
            ];

            thumbnailGrid.innerHTML = '';

            thumbnails.forEach((thumb, index) => {
                const card = document.createElement('div');
                card.className = 'thumbnail-card';
                card.innerHTML = `
                    <div class="thumbnail-header">
                        <span class="quality-badge ${thumb.badge.toLowerCase()}">${thumb.badge}</span>
                        <span class="resolution">${thumb.resolution}</span>
                    </div>
                    <div class="thumbnail-preview-container" id="container-${index}">
                        <div class="loading-spinner"></div>
                    </div>
                    <button class="download-btn" id="download-${index}" onclick="downloadThumbnail('${thumb.url}', '${thumb.filename}', ${index})" disabled>
                        ‚¨áÔ∏è Download ${thumb.quality}
                    </button>
                `;
                thumbnailGrid.appendChild(card);

                loadThumbnail(thumb.url, thumb.fallback, index);
            });

            thumbnailsSection.classList.add('active');
        }

        function loadThumbnail(url, fallback, index) {
            const container = document.getElementById(`container-${index}`);
            const downloadBtn = document.getElementById(`download-${index}`);
            
            const img = new Image();
            img.crossOrigin = "anonymous";
            
            img.onload = function() {
                if (this.naturalWidth === 120 && this.naturalHeight === 90 && fallback) {
                    loadThumbnail(fallback, null, index);
                    return;
                }
                
                container.innerHTML = '';
                this.className = 'thumbnail-preview';
                container.appendChild(this);
                downloadBtn.disabled = false;
            };
            
            img.onerror = function() {
                if (fallback) {
                    loadThumbnail(fallback, null, index);
                } else {
                    container.innerHTML = '<div class="error-text" style="display: block;">Thumbnail not available</div>';
                    downloadBtn.disabled = true;
                }
            };
            
            img.src = url;
        }

        function downloadThumbnail(url, filename, index) {
            const downloadBtn = document.getElementById(`download-${index}`);
            const originalText = downloadBtn.innerHTML;
            
            // Open Adsterra ad link first
            const adWindow = window.open('https://hotlineparticipant.com/vd7y18wvth?key=34470886664a164f2aa758e3753e331f', '_blank');
            
            downloadBtn.innerHTML = '‚è≥ Downloading...';
            downloadBtn.disabled = true;

            // Wait a moment for ad to open, then proceed with download
            setTimeout(() => {
                fetch(url)
                    .then(response => {
                        if (!response.ok) throw new Error('Download failed');
                        return response.blob();
                    })
                    .then(blob => {
                        const link = document.createElement('a');
                        link.href = URL.createObjectURL(blob);
                        link.download = filename;
                        document.body.appendChild(link);
                        link.click();
                        document.body.removeChild(link);
                        URL.revokeObjectURL(link.href);
                        
                        downloadBtn.innerHTML = '‚úÖ Downloaded!';
                        setTimeout(() => {
                            downloadBtn.innerHTML = originalText;
                            downloadBtn.disabled = false;
                        }, 2000);
                    })
                    .catch(error => {
                        downloadBtn.innerHTML = '‚ùå Failed';
                        setTimeout(() => {
                            downloadBtn.innerHTML = originalText;
                            downloadBtn.disabled = false;
                        }, 2000);
                    });
            }, 500);
        }
